Perceba que a leitura é feita antes do Enquanto para ser lida uma vez só e não cada vez que roda. [Engraçado pensar que isso parecia confuso no começo do aprendizado. Talvez tenha sido muita informação nova em em pouco tempo.]

algoritmo "somadorNumerico"
var
    cont, N, S, maior: Inteiro
inicio
    cont <- 1
    S <- 0
    Enquanto (cont <= 5) faca
            Escreval("Digite o", cont, "o. valor:")
            Leia(N)
            Se (N > maior) entao
                maior <- N
            FimSe
            S <- S + N
            cont <- cont + 1
    FimEnquanto
    Escreval("A soma de todos os valores foi", S)
    Escreval("O maior valor digitado foi", maior)

fimalgoritmo
