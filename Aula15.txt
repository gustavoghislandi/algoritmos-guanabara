[Lembrando que matrizes bidimensionais são sempre linha x coluna, nessa ordem]

    var
        m: vetor[1..3,1..2] de Inteiro
        l,c: Inteiro
    inicio
        para l <- 1 ate 3 faca
            para c <- 1 ate 2 faca
                Leia(m[l,c])
            fimpara
        fimpara
